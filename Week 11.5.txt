try:
    n=int(input())
    if(n<0):
        print("Error: Please enter a valid age.")
    else:
        print("You are {} years old.".format(n))
except TypeError:
    print("Error: Please enter a valid age.")
except ValueError:
    print("Error: Please enter a valid age.")
except EOFError:
    print("Error: Please enter a valid age.")
