def common_characters(s1, s2, n):
    common = []
    for char in s1:
        if char in s2 and char not in common:
            common.append(char)
            if len(common) == n:
                break
    return ''.join(common)
if __name__=="__main__":
    s1 = input().strip()
    s2 = input().strip()
    n = int(input())
    result = common_characters(s1, s2, n)
    print(result)
